4. 核心代码草图

# graph.py — Send API 扇出

def fanout_to_agents(state: AgentState):
    """根据每个 sub_question 的 source_types 创建并行 Send"""
    sends = []
    for sq in state["research_plan"]["sub_questions"]:
        for st in sq.get("source_types", ["blog"]):  # 默认 blog
            sends.append(Send("research_agent", {
                "agent_profile": st,
                "sub_question": sq,
            }))
    return sends

def build_graph(llm=None):
    graph = StateGraph(AgentState)
    graph.add_node("plan", make_plan_node(llm))
    graph.add_node("research_agent", make_research_agent(llm))  # 统一节点
    graph.add_node("merge", make_merge_node(llm))
    graph.add_node("summary", make_summary_node(llm))
    graph.add_node("critique", make_critique_node(llm))
    graph.add_node("final", make_final_node(llm))

    graph.add_edge(START, "plan")
    graph.add_conditional_edges("plan", fanout_to_agents, path_map=["research_agent"])
    graph.add_edge("research_agent", "merge")   # Send自动barrier
    graph.add_edge("merge", "summary")
    # ... 其余不变
    return graph
