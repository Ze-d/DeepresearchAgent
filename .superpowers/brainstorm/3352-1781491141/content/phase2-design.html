2. 新增的核心逻辑

交叉验证 (Cross-Validation)

# deepresearch/evidence/cross_validate.py  (新增)

def cross_validate_evidences(all_evidences: list[dict]) -> list[dict]:
    """
    比较来自不同 Agent 的 evidence:
    1. 语义聚类 — LLM 判断哪些 evidence 在讨论同一事实
    2. 一致性打分 — 同一聚类内的 evidence 互相确认 or 矛盾
    3. 调整 confidence:
       - 多 Agent 一致 → +0.1~0.2
       - 独有发现 → 标记 source_bias=True
       - 矛盾发现 → 标记 conflict=True, 降 confidence
    """
    # Step 1: 聚类 (LLM batch — 同 dedup 模式)
    clusters = _cluster_by_claim(evidences, llm)

    # Step 2: 每簇内打分
    for cluster in clusters:
        agents_in_cluster = {e["source_agent"] for e in cluster}
        if len(agents_in_cluster) >= 2:
            # 多 Agent 独立确认 → 提升 confidence
            for e in cluster:
                e["confidence"] = min(1.0, e["confidence"] + 0.1)
                e["cross_validated"] = True
                e["confirming_agents"] = list(agents_in_cluster)
        else:
            e = cluster[0]
            e["cross_validated"] = False
            e["source_bias"] = True  # 仅单一来源

冲突检测 (Conflict Detection)

def detect_conflicts(clusters: list[list[dict]], llm) -> list[dict]:
    """
    检测同一聚类内 evidence 之间的矛盾。
    返回 conflicts 列表，供 Critique 节点深度评审。
    """
    conflicts = []
    for cluster in clusters:
        if len(cluster) >= 2:
            # LLM 判断: 这些 evidence 互相矛盾吗？
            verdict = llm.invoke(CONFLICT_CHECK_PROMPT.format(
                claims="\n".join(f"[{e['source_agent']}] {e['claim']}" for e in cluster)
            ))
            if verdict["conflict"]:
                conflicts.append({
                    "topic": cluster[0]["claim"][:100],
                    "positions": {e["source_agent"]: e["claim"] for e in cluster},
                    "severity": verdict["severity"],  # "minor" | "major"
                })
    return conflicts

3. MergeSummary 数据模型（新增）

class MergeSummary(BaseModel):
    total_sources: int
    total_evidences: int
    cross_validated_count: int          # 被多个 Agent 确认的 evidence 数
    unique_findings_per_agent: dict     # {"paper": 5, "github": 3, ...}
    conflicts: list[dict]               # 矛盾列表
    source_bias_warnings: list[str]     # 单一来源偏见警告
    coverage_gaps: list[str]            # 覆盖不足的维度
