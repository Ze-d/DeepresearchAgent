# V2.1 Multi-Agent Concurrent Research — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the single-threaded `research_node` with 4 concurrent agents (paper/github/blog/docs) using LangGraph Send API fan-out, add cross-validation merge, and add optional human-in-the-loop review.

**Architecture:** LangGraph Send API fans out sub_questions to a unified `research_agent` node with per-agent `AgentProfile` injection. A new `merge` node collects parallel outputs. Phase 2 upgrades merge to cross-validate. Phase 3 adds `interrupt()`-based human review between merge and summary.

**Tech Stack:** Python 3.11+, LangGraph (Send API, interrupt), LangChain, Pydantic, FastAPI, Vue 3

**Spec:** [v2.1 Multi-Agent Design](../specs/2026-06-15-v2.1-multi-agent-design.md)

```python
"""V2.1: Unified Research Agent with AgentProfile strategy injection.

Each agent instance shares the same node function but differs in:
- search_modifiers: appended to search queries (e.g., site:arxiv.org)
- system_prompt: agent-specific extraction instructions
- evidence_instruction: what to look for in source content
"""

import json
import logging
import re
import uuid
from dataclasses import dataclass
from typing import Any

from langchain_core.language_models import BaseChatModel
from langchain_core.messages import SystemMessage

from deepresearch.state import AgentState
from deepresearch.tools import search_web, fetch_content
from deepresearch.config import Settings

logger = logging.getLogger(__name__)

# Rich console for real-time progress
_console = None


def _get_console():
    global _console
    if _console is None:

        _console = Console()
    return _console

# ── Evidence Extraction ──────────────────────────────────────


def _extract_evidences(profile: AgentProfile, sub_question: str,
                       source_content: str, llm: BaseChatModel) -> list[dict]:
    """用 LLM 从 source content 中抽取 evidence 列表，使用 agent profile 的指令。"""
    prompt = f"""你是一个{profile.name}。
{profile.system_prompt}

当前研究问题：
{sub_question}

资料内容：
{source_content}

任务：
请从资料中抽取可以支持研究报告的 evidence。

要求：
1. {profile.evidence_instruction}
2. 只抽取与当前问题相关的内容。
3. 不要编造资料中不存在的信息。
4. 每条 evidence 包含 claim、quote、confidence (0-1)。
5. 如果资料无关，返回空列表。
6. 只输出 JSON。

JSON 格式：
{{
  "evidences": [
    {{
      "claim": "...",
      "quote": "...",
      "confidence": 0.85
    }}
  ]
}}"""
    response = llm.invoke([SystemMessage(content=prompt)])
    raw = str(response.content) if hasattr(response, "content") else str(response)

    match = re.search(r"```(?:json)?\s*\n?(.*?)\n?```", raw, re.DOTALL)
    if match:
        raw = match.group(1).strip()

    try:
        data = json.loads(raw)
        return data.get("evidences", [])
    except json.JSONDecodeError:
        logger.warning("Failed to parse evidence JSON for agent %s: %s", profile.name, raw[:200])
        return []


# ── Node Factory ──────────────────────────────────────────────


def make_research_agent(llm: BaseChatModel):
    """创建 research_agent 节点（闭包注入 LLM）。

    该节点期望 state 中包含:
    - agent_profile: str — profile key（"paper"|"github"|"blog"|"docs"）
    - sub_question: dict — 要研究的 sub_question

    返回的 dict 会被 LangGraph Send API 自动合并到父 state 中。
    """
    def research_agent(state: AgentState) -> dict:
        profile_key = state.get("agent_profile", "blog")
        profile = AGENT_PROFILES.get(profile_key, AGENT_PROFILES["blog"])
        sq = state.get("sub_question", {})
        question = sq.get("question", state.get("user_query", ""))

        cfg = Settings()
        all_sources: list[dict] = []
        all_evidences: list[dict] = []

        queries = sq.get("search_queries", [])
        _get_console().print(f"   [{profile.name}] 开始研究: {question[:60]}...")

        for idx, query in enumerate(queries[:2]):
            # 构建增强搜索查询：原始 query + profile modifiers
            enhanced_query = f"{query} {' '.join(profile.search_modifiers[:2])}"
            _get_console().print(f"   [{profile.name}] 🔎 搜索 ({idx + 1}/{min(len(queries), 2)}): {enhanced_query[:80]}")
            results = search_web(enhanced_query, max_results=cfg.max_search_results)
            _get_console().print(f"   [{profile.name}]    找到 {len(results)} 条结果")

            for r in results:
                source_id = str(uuid.uuid4())[:8]
                source_dict = {
                    "id": source_id,
                    "title": r.title,
                    "url": r.url,
                    "snippet": r.snippet,
                    "content": None,
                    "source_type": profile_key,
                    "source_agent": profile_key,
                    "score": 0.5,
                }
                content = fetch_content(r.url)
                if content:
                    source_dict["content"] = content
                    evidences = _extract_evidences(profile, question, content, llm)
                    for ev in evidences:
                        ev["id"] = str(uuid.uuid4())[:8]
                        ev["source_id"] = source_id
                        ev["source_agent"] = profile_key
                        all_evidences.append(ev)

                all_sources.append(source_dict)

        logger.info("[%s] Done: %d sources, %d evidences", profile.name, len(all_sources), len(all_evidences))

        return {
            "search_results": state.get("search_results", []) + [
                {"query": q, "source_agent": profile_key}
                for q in queries[:2]
            ],
            "sources": state.get("sources", []) + all_sources,
            "evidences": state.get("evidences", []) + all_evidences,
            "agent_profile": profile_key,
        }

    return research_agent


# Backward-compatible module-level alias
def research_agent(state: AgentState) -> dict:
    from deepresearch.llm import build_llm
    llm = build_llm()
    return make_research_agent(llm)(state)
```

```python
"""V2.1 Merge Node — collects parallel agent outputs, deduplicates, and ranks.

Phase 1: Simple collection + URL dedup + semantic evidence dedup + source ranking.
Phase 2: Upgraded to three-stage pipeline with cross-validation.
"""

import logging
from typing import Any

from langchain_core.language_models import BaseChatModel

from deepresearch.state import AgentState
from deepresearch.config import Settings

logger = logging.getLogger(__name__)


def _dedup_sources_by_url(sources: list[dict]) -> list[dict]:
    """按 URL 精确去重 sources，保留首次出现。"""
    seen: set[str] = set()
    deduped: list[dict] = []
    for s in sources:
        url = s.get("url", "")
        if url and url not in seen:
            seen.add(url)
            deduped.append(s)
        elif not url:
            deduped.append(s)
    return deduped


def make_merge_node(llm: BaseChatModel):
    """创建 merge_node（闭包注入 LLM）。"""

    def merge_node(state: AgentState) -> dict:
        cfg = Settings()
        sources = list(state.get("sources", []))
        evidences = list(state.get("evidences", []))

        logger.info("Merge: collecting %d sources, %d evidences", len(sources), len(evidences))

        # Stage 1: Collect & Dedup sources by URL
        sources = _dedup_sources_by_url(sources)
        logger.info("Merge after URL dedup: %d sources", len(sources))

        # Stage 1: Evidence semantic dedup
        if evidences and cfg.dedup_enabled:
            from deepresearch.evidence.dedup import deduplicate_evidences
            evidences = deduplicate_evidences(evidences, llm)
            logger.info("Merge after evidence dedup: %d evidences", len(evidences))

        # Stage 1: Source ranking
        if sources and cfg.source_ranking_enabled:
            from deepresearch.evidence.ranking import rank_sources
            sources = rank_sources(sources)

        # Generate simple merge summary
        agent_counts: dict[str, int] = {}
        for s in sources:
            agent = s.get("source_agent", "unknown")
            agent_counts[agent] = agent_counts.get(agent, 0) + 1

        merge_summary = {
            "total_sources": len(sources),
            "total_evidences": len(evidences),
            "unique_findings_per_agent": agent_counts,
            "cross_validated_count": 0,   # Phase 2 fills this
            "conflicts": [],               # Phase 2 fills this
            "source_bias_warnings": [],     # Phase 2 fills this
            "coverage_gaps": [],            # Phase 2 fills this
        }

        logger.info("Merge complete: %d sources, %d evidences, agents=%s",
                    len(sources), len(evidences), list(agent_counts.keys()))

        return {
            "sources": sources,
            "evidences": evidences,
            "merge_summary": merge_summary,
            "status": "merged",
        }

    return merge_node
```

```python
import logging

from langchain_core.language_models import BaseChatModel
from langgraph.graph import StateGraph, START, END
from langgraph.types import Send

from deepresearch.state import AgentState
from deepresearch.nodes import (
    make_plan_node,
    make_research_agent,
    make_merge_node,
    make_summary_node,
    make_critique_node,
    make_final_node,
)
from deepresearch.llm import build_llm

logger = logging.getLogger(__name__)


# ── Routing Functions ────────────────────────────────────────


def route_after_critique(state: AgentState) -> str:
    critique = state.get("critique_result") or {}
    iteration = state.get("iteration", 0)
    max_iterations = state.get("max_iterations", 2)

    if critique.get("pass") is True:
        logger.info("Critique passed (iteration %d/%d) → routing to final", iteration, max_iterations)
        return "final"
    if iteration >= max_iterations:
        logger.info("Max iterations reached (%d/%d) → routing to final", iteration, max_iterations)
        return "final"
    logger.info("Critique not passed (iteration %d/%d) → continuing research", iteration, max_iterations)
    return "research_agent"


def route_after_plan(state: AgentState) -> str:
    if state.get("status") == "error" or not state.get("research_plan"):
        logger.info("Planning failed → ending workflow")
        return "end"
    logger.info("Planning succeeded → routing to research_agent")
    return "research_agent"


# ── Fan-Out to Research Agents ───────────────────────────────


def fanout_to_agents(state: AgentState) -> list[Send]:
    """根据每个 sub_question 的 source_types 创建并行 Send。

    每个 Send 将单个 sub_question 和 agent_profile 注入 state，
    research_agent 节点对每个 Send 独立执行。
    """
    plan = state.get("research_plan") or {}
    sub_questions = plan.get("sub_questions", [])

    # Handle critique follow-up: route to all agents
    critique = state.get("critique_result") or {}
    if critique.get("pass") is False and critique.get("new_search_queries"):
        follow_up_sq = {
            "id": "critique_followup",
            "question": state.get("user_query", ""),
            "priority": 1,
            "search_queries": critique["new_search_queries"],
            "source_types": ["paper", "github", "blog", "docs"],
        }
        sub_questions = [follow_up_sq]

    sends: list[Send] = []
    for sq in sub_questions:
        source_types = sq.get("source_types", ["blog"])
        for st in source_types:
            sends.append(Send(
                "research_agent",
                {
                    "agent_profile": st,
                    "sub_question": sq,
                },
            ))
    if not sends:
        # Fallback: at least one agent
        sends.append(Send("research_agent", {
            "agent_profile": "blog",
            "sub_question": {
                "id": "default", "question": state["user_query"],
                "priority": 1, "search_queries": [state["user_query"]],
                "source_types": ["blog"],
            },
        }))
    logger.info("Fan-out: %d Send(s) to research_agent", len(sends))
    return sends


# ── Graph Builder ─────────────────────────────────────────────


def build_graph(llm: BaseChatModel | None = None) -> StateGraph:
    """构建 V2.1 DeepResearch Agent StateGraph。

    Topology:
        START → plan → research_agent × N (concurrent) → merge
        → summary → critique → {research_agent | final} → END
    """
    if llm is None:
        llm = build_llm()

    logger.info("Building V2.1 StateGraph: plan → research_agent(×N) → merge → summary → critique")

    graph = StateGraph(AgentState)

    graph.add_node("plan", make_plan_node(llm))
    graph.add_node("research_agent", make_research_agent(llm))
    graph.add_node("merge", make_merge_node(llm))
    graph.add_node("summary", make_summary_node(llm))
    graph.add_node("critique", make_critique_node(llm))
    graph.add_node("final", make_final_node(llm))

    graph.add_edge(START, "plan")

    # Plan → research_agent fan-out via Send API
    graph.add_conditional_edges(
        "plan",
        fanout_to_agents,
        path_map=["research_agent"],  # all Sends go to same node
    )

    # research_agent → merge (Send API auto-barrier: all agents complete before merge)
    graph.add_edge("research_agent", "merge")

    # merge → summary → critique
    graph.add_edge("merge", "summary")
    graph.add_edge("summary", "critique")

    # critique → final or loop back to research_agent
    graph.add_conditional_edges(
        "critique",
        route_after_critique,
        {
            "research_agent": "research_agent",
            "final": "final",
        },
    )

    graph.add_edge("final", END)

    logger.info("V2.1 StateGraph built: %d nodes", len(graph.nodes))
    return graph
```

```python
        with DDGS() as ddgs:
            raw = list(ddgs.text(search_query, max_results=max_results))

        logger.warning("DuckDuckGo search failed for: %s", search_query)
        return []

    results = []
    for item in raw:
        results.append(SearchResult(
            title=item.get("title", ""),
            url=item.get("href", ""),
            snippet=item.get("body", ""),
        ))
    return results
```

```python
    for multi-agent confirmed findings, marks solo findings for bias review.
detect_conflicts: identifies contradictory claims within clusters via LLM.
"""

import logging
from typing import Any

from langchain_core.language_models import BaseChatModel
from langchain_core.messages import SystemMessage

logger = logging.getLogger(__name__)

_DUP_CHECK_PROMPT = """判断以下两条 evidence 是否表达相同的核心信息（即使措辞不同）。

Evidence A:
  claim: "{claim_a}"
  agent: {agent_a}

Evidence B:
  claim: "{claim_b}"
  agent: {agent_b}

只回答 YES 或 NO。"""

_CONFLICT_PROMPT = """判断以下 evidence 之间是否存在实质性矛盾（不是措辞差异）。

Claims:
{claims_text}

如果存在实质性矛盾（相反结论、冲突的数据、矛盾的建议），返回：
{{"conflict": true, "severity": "major"|"minor"}}

如果不存在矛盾，返回：
{{"conflict": false, "severity": "none"}}"""


def _are_same_claim(ev_a: dict, ev_b: dict, llm: BaseChatModel) -> bool:
    """LLM 判断两条 evidence 是否在讨论同一事实。"""
    prompt = _DUP_CHECK_PROMPT.format(
        claim_a=ev_a.get("claim", ""),
        agent_a=ev_a.get("source_agent", "unknown"),
        claim_b=ev_b.get("claim", ""),
        agent_b=ev_b.get("source_agent", "unknown"),
    )
    try:
        response = llm.invoke([SystemMessage(content=prompt)])
        text = str(response.content).strip().upper() if hasattr(response, "content") else ""
        return "YES" in text
    except Exception:
        logger.warning("Cross-validate LLM call failed", exc_info=True)
        return False


def cross_validate_evidences(evidences: list[dict], llm: BaseChatModel) -> list[dict]:
    """跨 Agent 交叉验证：聚类 + 一致性调整 confidence。

    Returns 修改后的 evidences（新增 cross_validated, source_bias 字段）。
    """
    if not evidences or len(evidences) < 2:
        for ev in evidences:
            ev["cross_validated"] = False
            ev["source_bias"] = True
        return evidences

    n = len(evidences)
    clusters: list[list[int]] = []  # list of index groups
    assigned: set[int] = set()

    # Simple O(n²) clustering — same claim → same cluster
    for i in range(n):
        if i in assigned:
            continue
        cluster = [i]
        for j in range(i + 1, n):
            if j in assigned:
                continue
            if _are_same_claim(evidences[i], evidences[j], llm):
                cluster.append(j)
                assigned.add(j)
        assigned.add(i)
        clusters.append(cluster)

    logger.info("Cross-validate: %d evidences → %d clusters", n, len(clusters))

    # Adjust confidence per cluster
    for cluster in clusters:
        indices = cluster
        agents_in_cluster = set()
        for idx in indices:
            agent = evidences[idx].get("source_agent", "unknown")
            agents_in_cluster.add(agent)

        if len(agents_in_cluster) >= 2:
            # Multi-agent confirmed: boost confidence
            boost = min(0.2, 0.05 * len(agents_in_cluster))
            for idx in indices:
                old_conf = evidences[idx].get("confidence", 0)
                evidences[idx]["confidence"] = min(1.0, old_conf + boost)
                evidences[idx]["cross_validated"] = True
                evidences[idx]["confirming_agents"] = list(agents_in_cluster)
        else:
            # Solo finding
            for idx in indices:
                evidences[idx]["cross_validated"] = False
                evidences[idx]["source_bias"] = True

    return evidences


def detect_conflicts(clusters: list[list[dict]], llm: BaseChatModel) -> list[dict]:
    """检测聚类内的 evidence 矛盾。

    Args:
        clusters: 由 cross_validate_evidences 输出的聚类。
        llm: LLM 实例。

    Returns:
        conflicts 列表，每个 conflict 含 topic, positions, severity。
    """
    conflicts: list[dict] = []
    for cluster in clusters:
        if len(cluster) < 2:
            continue

        agents = {e.get("source_agent", "unknown") for e in cluster}
        if len(agents) < 2:
            continue  # 同一 agent 不可能矛盾 — 已在 dedup 阶段去除

        claims_text = "\n".join(
            f"[{e.get('source_agent', '?')}] {e.get('claim', '')[:200]}"
            for e in cluster
        )
        prompt = _CONFLICT_PROMPT.format(claims_text=claims_text)

        try:
            import json
            response = llm.invoke([SystemMessage(content=prompt)])
            raw = str(response.content) if hasattr(response, "content") else str(response)
            verdict = json.loads(raw)
        except Exception:
            logger.warning("Conflict detection LLM call failed", exc_info=True)
            continue

        if verdict.get("conflict"):
            conflicts.append({
                "topic": cluster[0].get("claim", "")[:100],
                "positions": {e.get("source_agent", ""): e.get("claim", "")
                              for e in cluster},
                "severity": verdict.get("severity", "minor"),
            })

    if conflicts:
        logger.info("Detected %d cross-agent conflicts", len(conflicts))
    return conflicts
```

```python
# deepresearch/nodes/merge.py — replace merge_node function:

def make_merge_node(llm: BaseChatModel):
    def merge_node(state: AgentState) -> dict:
        cfg = Settings()
        sources = list(state.get("sources", []))
        evidences = list(state.get("evidences", []))

        logger.info("Merge Phase 2: collecting %d sources, %d evidences",
                    len(sources), len(evidences))

        # ═══ Stage 1: Collect & Normalize ═══
        # Ensure every evidence has source_agent
        for ev in evidences:
            if "source_agent" not in ev:
                ev["source_agent"] = "unknown"

        # ═══ Stage 2: Dedup & Cross-Validate ═══

        # 2a. Source URL dedup
        sources = _dedup_sources_by_url(sources)

        # 2b. Evidence semantic dedup (within same source)
        if evidences and cfg.dedup_enabled:
            from deepresearch.evidence.dedup import deduplicate_evidences
            evidences = deduplicate_evidences(evidences, llm)

        # 2c. Cross-validation (Phase 2)
        if evidences and len(set(e.get("source_agent") for e in evidences)) >= 2:
            from deepresearch.evidence.cross_validate import (
                cross_validate_evidences, detect_conflicts
            )
            evidences = cross_validate_evidences(evidences, llm)
            logger.info("Cross-validation complete for %d evidences", len(evidences))

            # 2d. Conflict detection
            # Group evidences by (approximate) claim clusters
            # Reuse cross_validate's clustering output via evidence metadata
            conflicts = detect_conflicts(
                _build_clusters_from_evidence(evidences), llm
            )
        else:
            conflicts = []

        # 2e. Source ranking
        if sources and cfg.source_ranking_enabled:
            from deepresearch.evidence.ranking import rank_sources
            sources = rank_sources(sources)

        # ═══ Stage 3: Quality Report ═══
        agent_counts: dict[str, int] = {}
        for s in sources:
            agent = s.get("source_agent", "unknown")
            agent_counts[agent] = agent_counts.get(agent, 0) + 1

        cross_validated = sum(1 for ev in evidences if ev.get("cross_validated"))
        solo_findings = sum(1 for ev in evidences if ev.get("source_bias"))
        source_bias_warnings = []
        if solo_findings > len(evidences) * 0.5:
            source_bias_warnings.append(
                f"{solo_findings}/{len(evidences)} 条证据仅来自单一来源，可能存在来源偏见"
            )

        merge_summary = {
            "total_sources": len(sources),
            "total_evidences": len(evidences),
            "cross_validated_count": cross_validated,
            "unique_findings_per_agent": agent_counts,
            "conflicts": conflicts,
            "source_bias_warnings": source_bias_warnings,
            "coverage_gaps": [],  # Reserved for future enhancement
        }

        logger.info("Merge Phase 2 complete: %d src, %d ev, %d xv, %d conflicts",
                    len(sources), len(evidences), cross_validated, len(conflicts))

        return {
            "sources": sources,
            "evidences": evidences,
            "merge_summary": merge_summary,
            "status": "merged",
        }

    return merge_node


def _build_clusters_from_evidence(evidences: list[dict]) -> list[list[dict]]:
    """从 evidence 的 cross_validated/confirming_agents 字段重建聚类。

    所有 cross_validated=True 且 confirming_agents 重叠的 evidence 属于同一个聚类。
    其余每一条形成一个单元素聚类。
    """
    # Simple: group by confirming_agents overlap
    validated: list[dict] = [e for e in evidences if e.get("cross_validated")]
    solo: list[dict] = [e for e in evidences if not e.get("cross_validated")]

    clusters: list[list[dict]] = []
    used: set[int] = set()

    for i, ev in enumerate(validated):
        if i in used:
            continue
        cluster = [ev]
        agents_i = set(ev.get("confirming_agents", []))
        for j in range(i + 1, len(validated)):
            if j in used:
                continue
            agents_j = set(validated[j].get("confirming_agents", []))
            if agents_i & agents_j:
                cluster.append(validated[j])
                used.add(j)
        used.add(i)
        clusters.append(cluster)

    # Solo findings: each in its own cluster
    for ev in solo:
        clusters.append([ev])

    return clusters
```

```python
# deepresearch/nodes/human_review.py
"""V2.1 Phase 3: Human-in-the-Loop review node using LangGraph interrupt().

When enabled, pauses workflow execution after merge and waits for human input
via POST /api/tasks/{id}/review. Three possible decisions:
- approve: continue to critique → final
- amend: inject new search queries and loop back to research_agent
- redo: revise research plan and restart from plan (or research_agent)
"""

import logging
from typing import Any

from langchain_core.language_models import BaseChatModel
from langgraph.types import interrupt

from deepresearch.state import AgentState
from deepresearch.config import settings

logger = logging.getLogger(__name__)


def make_human_review_node(llm: BaseChatModel):
    """创建 human_review 节点（闭包注入 LLM）。

    Uses LangGraph interrupt() to pause execution. The graph resumes
    when external code calls graph.stream() or graph.invoke() with
    the `Command(resume=...)` mechanism.
    """
    def human_review_node(state: AgentState) -> dict:
        # If human review is disabled globally, auto-approve
        if not settings.human_review_enabled:
            logger.debug("Human review disabled — auto-approving")
            return {
                "human_review": {"action": "approved", "notes": "auto: human review disabled"},
                "status": "reviewed_approved",
            }

        merge_summary = state.get("merge_summary", {})
        sources = state.get("sources", [])
        evidences = state.get("evidences", [])
        conflicts = merge_summary.get("conflicts", [])

        logger.info("Human review required: %d sources, %d evidences, %d conflicts",
                    len(sources), len(evidences), len(conflicts))

        # Pause the graph and present review material to the human
        human_decision = interrupt({
            "action": "review_requested",
            "task_id": state.get("task_id", "unknown"),
            "merge_summary": {
                "total_sources": merge_summary.get("total_sources", len(sources)),
                "total_evidences": merge_summary.get("total_evidences", len(evidences)),
                "cross_validated_count": merge_summary.get("cross_validated_count", 0),
                "unique_findings_per_agent": merge_summary.get("unique_findings_per_agent", {}),
                "conflicts": conflicts,
                "source_bias_warnings": merge_summary.get("source_bias_warnings", []),
                "coverage_gaps": merge_summary.get("coverage_gaps", []),
            },
            "sources": [
                {"id": s.get("id"), "title": s.get("title"), "url": s.get("url"),
                 "source_agent": s.get("source_agent", "unknown"), "score": s.get("score", 0)}
                for s in sources[:30]  # Limit to avoid overwhelming
            ],
            "evidences": [
                {"id": e.get("id"), "claim": e.get("claim"),
                 "confidence": e.get("confidence", 0),
                 "source_agent": e.get("source_agent", "unknown"),
                 "cross_validated": e.get("cross_validated", False)}
                for e in evidences[:50]
            ],
            "prompt": "请审核研究结果，选择 approve / amend / redo",
        })

        # human_decision comes from POST /api/tasks/{id}/review
        action = human_decision.get("action", "approve")
        logger.info("Human review decision: %s", action)

        if action == "amend":
            return {
                "human_review": human_decision,
                "critique_result": {
                    "pass": False,
                    "overall_score": 0.0,
                    "dimensions": {
                        "fact_check": {"score": 0.0, "issues": [], "status": "pass"},
                        "logic_coherence": {"score": 0.0, "issues": [], "status": "pass"},
                        "coverage": {"score": 0.0, "issues": [], "status": "pass"},
                    },
                    "issues": [{
                        "type": "human_amendment",
                        "severity": "high",
                        "description": human_decision.get("notes", "Human requested amendments"),
                        "suggested_action": "execute new search queries",
                    }],
                    "new_search_queries": human_decision.get("new_queries", []),
                },
                "status": "reviewed_amend",
            }
        elif action == "redo":
            return {
                "human_review": human_decision,
                "status": "reviewed_redo",
                "errors": [f"Human requested redo: {human_decision.get('notes', '')}"],
            }
        else:  # approve (default)
            return {
                "human_review": human_decision,
                "status": "reviewed_approved",
            }

    return human_review_node
```

```python
    new_queries: list[str] = []


@router.post("/tasks/{task_id}/review")
def submit_review(task_id: str, req: ReviewRequest):
    """提交人工审核结果，恢复被 interrupt 的图执行。

    当任务配置了 human_review=true 时，
    图会在 merge 完成后暂停，等待此端点提交审核决策。

    Request Body:
        action: "approve" | "amend" | "redo"
        notes: 审核备注
        new_queries: amend 时的补充搜索查询列表
    """
    task = task_manager.get(task_id)
    if task is None:
        raise HTTPException(status_code=404, detail="Task not found")

    if task["status"] != TaskStatus.REVIEW_REQUIRED.value:
        raise HTTPException(status_code=409, detail="Task is not in review required state")

    decision = {
        "action": req.action,
        "notes": req.notes,
        "new_queries": req.new_queries if req.action == "amend" else [],
    }

    from langgraph.types import Command
    # Resume graph execution with human decision
    # Note: graph_manager needs to store the graph reference per task
    if task_id in _graph_sessions:
        graph, config = _graph_sessions[task_id]
        # Resume with Command(resume=decision)
        _resume_graph(task_id, graph, config, decision)
        return {"status": "resumed", "action": req.action}
    else:
        raise HTTPException(status_code=404, detail="Graph session not found for task")
```

```python
class TestHumanInTheLoop:
    def test_hitl_review_endpoint(self, client, monkeypatch):
        """启用 human_review 时任务暂停等待 /review API"""

        task_id = resp.json()["task_id"]

        import time
        # Poll until review_required state
        for _ in range(60):
            resp = client.get(f"/api/tasks/{task_id}")
            data = resp.json()
            if data["status"] == "review_required":

            time.sleep(0.3)

        # Task should be in review_required state

        # Submit approve review
        resp = client.post(f"/api/tasks/{task_id}/review", json={
```
